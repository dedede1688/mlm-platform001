**业务规则设计**：

支持 **4 种券类型**（行业通用）：

| 类型 | 字段 | 业务规则 |
|------|------|---------|
| **满减券** | `minAmount` + `discountAmount` | 订单金额 ≥ minAmount 可用，立减 discountAmount |
| **折扣券** | `discountRate` (0.7 = 7折) + `maxDiscount` | 订单 × discountRate，最高减 maxDiscount |
| **新人券** | `discountAmount` + `newUserOnly` | 仅注册 < 7 天的用户可用 |
| **限时券** | `validFrom` + `validTo` | 在时间窗口内可用 |

**两张表设计**：

- `Coupon` —— 优惠券**模板**（admin 创建/管理）
- `UserCoupon` —— 用户**领取记录**（用户领券后写入）

## Step 1：Schema + Migration

### 1.1 Coupon model 设计

**位置**：`prisma/schema.prisma`（建议放在 SystemConfig 之后）

```prisma
model Coupon {
  id              String    @id @default(uuid())
  code            String    @unique
  name            String
  type            String    // discount(满减) / rate(折扣) / new_user(新人) / time_limited(限时)
  // 满减券字段（type='discount'）
  minAmount       Decimal?  @map("min_amount") @db.Decimal(10, 2)    // 满 X 元可用
  discountAmount  Decimal?  @map("discount_amount") @db.Decimal(10, 2) // 减 Y 元
  // 折扣券字段（type='rate'）
  discountRate    Decimal?  @map("discount_rate") @db.Decimal(3, 2)   // 0.70 = 7 折
  maxDiscount     Decimal?  @map("max_discount") @db.Decimal(10, 2)  // 最高减 Y 元
  // 通用字段
  totalQuantity   Int       @default(0) @map("total_quantity")    // 发行总量
  usedQuantity    Int       @default(0) @map("used_quantity")     // 已使用
  perUserLimit    Int       @default(1) @map("per_user_limit")    // 每人限领
  validFrom       DateTime  @map("valid_from")
  validTo         DateTime  @map("valid_to")
  status          String    @default("active")                    // active / disabled / expired
  description     String?   @db.Text
  createdAt       DateTime  @default(now()) @map("created_at")
  updatedAt       DateTime  @updatedAt @map("updated_at")

  userCoupons     UserCoupon[]

  @@index([status])
  @@index([type])
  @@index([validFrom, validTo])
  @@index([code])
  @@map("coupons")
}
```

**关键设计决策**：
- **4 种 type 用一个表 + 可空字段**——避免 4 张表的 JOIN，admin 操作简单
- **`Decimal` 类型**——金额精度（避免浮点误差）
- **status 字段**——`active` / `disabled`（admin 手动）/ `expired`（cron 自动，超出 validTo 后由 daily-tasks 标记）
- **code 唯一索引**——用户输入 code 领券（v55.2 实现）
- **总/已用 数量**——admin 可以看到发行量 + 已用进度

### 1.2 UserCoupon model 设计

```prisma
model UserCoupon {
  id            String    @id @default(uuid())
  userId        String    @map("user_id")
  couponId      String    @map("coupon_id")
  status        String    @default("unused")  // unused / used / expired
  usedAt        DateTime? @map("used_at")
  usedOrderId   String?   @map("used_order_id")
  claimedAt     DateTime  @default(now()) @map("claimed_at")

  user          User      @relation(fields: [userId], references: [id], onDelete: Cascade)
  coupon        Coupon    @relation(fields: [couponId], references: [id], onDelete: Cascade)

  @@unique([userId, couponId])   // 同一券一人只能领一次（或由 perUserLimit 控制）
  @@index([userId, status])
  @@index([couponId])
  @@index([claimedAt])
  @@map("user_coupons")
}
```

## Step 2：CouponService 后端逻辑

### 2.1 文件结构

**位置**：`src/lib/services/coupon.service.ts`（新建）

```typescript
import { prisma } from '@/lib/prisma'
import { logger } from '@/lib/logger'
import { logOperation } from '@/lib/utils/operation-log'

export interface CreateCouponParams {
  code: string
  name: string
  type: 'discount' | 'rate' | 'new_user' | 'time_limited'
  minAmount?: number
  discountAmount?: number
  discountRate?: number
  maxDiscount?: number
  totalQuantity: number
  perUserLimit?: number
  validFrom: Date
  validTo: Date
  description?: string
}

export class CouponService {
  // 列表（admin）
  static async list(params: {
    page: number
    limit: number
    type?: string
    status?: string
    keyword?: string
  }) {
    const { page, limit, type, status, keyword } = params
    const skip = (page - 1) * limit
    const where: any = {}
    if (type) where.type = type
    if (status) where.status = status
    if (keyword) {
      where.OR = [
        { code: { contains: keyword, mode: 'insensitive' } },
        { name: { contains: keyword, mode: 'insensitive' } },
      ]
    }
    const [coupons, total] = await Promise.all([
      prisma.coupon.findMany({ where, orderBy: { createdAt: 'desc' }, skip, take: limit }),
      prisma.coupon.count({ where }),
    ])
    return { coupons, total, totalPages: Math.ceil(total / limit) }
  }

  // 详情
  static async getById(id: string) {
    return prisma.coupon.findUnique({ where: { id } })
  }

  // 创建
  static async create(params: CreateCouponParams) {
    // 校验
    CouponService.validateParams(params)
    return prisma.coupon.create({ data: params })
  }

  // 更新
  static async update(id: string, params: Partial<CreateCouponParams>) {
    return prisma.coupon.update({ where: { id }, data: params })
  }

  // 删除（软删除？硬删除？）
  // 业务规则：如果已发放 user_coupons，禁止硬删除
  // 方案：只允许 status='disabled'，不直接 DELETE
  static async disable(id: string) {
    return prisma.coupon.update({
      where: { id },
      data: { status: 'disabled' },
    })
  }

  // 启用
  static async enable(id: string) {
    return prisma.coupon.update({
      where: { id },
      data: { status: 'active' },
    })
  }

  // 校验参数
  private static validateParams(params: CreateCouponParams) {
    if (!params.code || params.code.length < 4) {
      throw new Error('券码至少 4 个字符')
    }
    if (!params.name) {
      throw new Error('券名称不能为空')
    }
    if (params.validTo <= params.validFrom) {
      throw new Error('失效时间必须晚于生效时间')
    }
    if (params.type === 'discount' && (!params.minAmount || !params.discountAmount)) {
      throw new Error('满减券必须设置满 X 减 Y')
    }
    if (params.type === 'rate' && (!params.discountRate || !params.maxDiscount)) {
      throw new Error('折扣券必须设置折扣率和最高减额')
    }
    if (params.totalQuantity < 1) {
      throw new Error('发行数量必须 ≥ 1')
    }
  }
}
```

## Step 3：Admin API 4 个路由

### 3.1 `/api/admin/coupons` —— GET 列表 + POST 创建

**参考模式**：`src/app/api/admin/withdrawal-templates/route.ts`（已经 read，CRUD 模式相同）

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { verifyPermission } from '@/lib/utils/admin-auth'
import { CouponService } from '@/lib/services/coupon.service'

export async function GET(request: NextRequest) {
  try {
    const { error: authError } = await verifyPermission(request, ['super_admin', 'finance_admin', 'goods_admin'])
    if (authError) return authError

    const { searchParams } = new URL(request.url)
    const page = parseInt(searchParams.get('page') || '1')
    const limit = parseInt(searchParams.get('limit') || '20')
    const type = searchParams.get('type') || undefined
    const status = searchParams.get('status') || undefined
    const keyword = searchParams.get('keyword') || undefined

    const result = await CouponService.list({ page, limit, type, status, keyword })
    return NextResponse.json({ success: true, data: result })
  } catch (error) {
    console.error('Get coupons error:', error)
    return NextResponse.json({ success: false, message: '获取优惠券列表失败' }, { status: 500 })
  }
}

export async function POST(request: NextRequest) {
  try {
    const { user: admin, error: authError } = await verifyPermission(request, ['super_admin', 'finance_admin', 'goods_admin'])
    if (authError || !admin) return authError

    const body = await request.json()
    const coupon = await CouponService.create({
      code: body.code,
      name: body.name,
      type: body.type,
      minAmount: body.minAmount ? Number(body.minAmount) : undefined,
      discountAmount: body.discountAmount ? Number(body.discountAmount) : undefined,
      discountRate: body.discountRate ? Number(body.discountRate) : undefined,
      maxDiscount: body.maxDiscount ? Number(body.maxDiscount) : undefined,
      totalQuantity: Number(body.totalQuantity),
      perUserLimit: body.perUserLimit ? Number(body.perUserLimit) : 1,
      validFrom: new Date(body.validFrom),
      validTo: new Date(body.validTo),
      description: body.description,
    })

    await logOperation({
      userId: admin.id,
      action: 'CREATE',
      module: 'coupon',
      targetId: coupon.id,
      newValue: { code: coupon.code, name: coupon.name },
    })

    return NextResponse.json({ success: true, data: coupon, message: '优惠券创建成功' })
  } catch (error) {
    console.error('Create coupon error:', error)
    const message = error instanceof Error ? error.message : '创建优惠券失败'
    return NextResponse.json({ success: false, message }, { status: 400 })
  }
}
```

### 3.2 `/api/admin/coupons/[id]/route.ts` —— GET 详情 + PATCH 更新 + DELETE 删除

```typescript
export async function GET(request: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  // ... verifyPermission
  const { id } = await params
  const coupon = await CouponService.getById(id)
  if (!coupon) return NextResponse.json({ success: false, message: '优惠券不存在' }, { status: 404 })
  return NextResponse.json({ success: true, data: coupon })
}

export async function PATCH(request: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  // ... verifyPermission
  const { id } = await params
  const body = await request.json()
  const coupon = await CouponService.update(id, {
    name: body.name,
    totalQuantity: body.totalQuantity ? Number(body.totalQuantity) : undefined,
    validFrom: body.validFrom ? new Date(body.validFrom) : undefined,
    validTo: body.validTo ? new Date(body.validTo) : undefined,
    description: body.description,
  })
  // ... logOperation
  return NextResponse.json({ success: true, data: coupon })
}

export async function DELETE(request: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  // 不允许硬删除——改用 disable
  const { user: admin, error: authError } = await verifyPermission(request, ['super_admin'])
  if (authError || !admin) return authError
  const { id } = await params
  await CouponService.disable(id)
  // ... logOperation
  return NextResponse.json({ success: true, message: '优惠券已禁用' })
}
```

### 3.3 `/api/admin/coupons/[id]/toggle/route.ts` —— PATCH 启用/禁用

```typescript
export async function PATCH(request: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  // ... verifyPermission
  const { id } = await params
  const { action } = await request.json() // 'enable' | 'disable'
  const coupon = action === 'enable' 
    ? await CouponService.enable(id)
    : await CouponService.disable(id)
  return NextResponse.json({ success: true, data: coupon })
}
```

## 风险点

1. **Schema 字段命名不一致**：
   - 之前 v12 教训：camelCase 字段名 ≠ snake_case 列名（prisma 用 @map 映射）
   - 必须严格用 `@map("snake_case")` —— 派单里已正确

2. **Decimal 类型精度**：
   - Prisma 6 的 Decimal 默认精度可能不够
   - 必须用 `@db.Decimal(10, 2)` 显式指定（10 位数 + 2 位小数）

3. **删除策略**：
   - 如果 user_coupons 已存在记录，删除 coupon 会级联删除 user_coupons
   - 派单里用 `disable` 替代 DELETE——更安全

4. **migration 不自动跑**：
   - v46.3.2 教训：必须告诉胡子哥手动跑 migration

5. **admin-menu 角色权限**：
   - 派单让 goods_admin + finance_admin 都能管理优惠券
   - 业务上合理（券是营销工具）
   - 如果胡子哥想限定 super_admin，调整 ROLE_MENUS

6. **time_limited 类型冗余**：
   - `time_limited` 实际上只是 validFrom/validTo 较短
   - 可考虑合并到 `discount` 类型 + 时间字段
   - 派单保留 4 种——更清晰，admin 选择时不会混淆

7. **未触发 daily-tasks 自动 expire**：
   - cron daily-tasks 路由需要加 `expireCoupons` 函数
   - **v55.1 不做**——派单里 status='expired' 字段已预留，v55.4（？）再补
