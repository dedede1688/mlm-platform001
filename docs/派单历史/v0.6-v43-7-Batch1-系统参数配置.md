## 2. 统一配置工具 `src/lib/config/system-parameters.ts`（新文件）

**职责**：统一封装"读 / 写" `SystemConfig` 表的业务配置项。

**API 设计**：

```typescript
// src/lib/config/system-parameters.ts

import { prisma } from '@/lib/prisma'
import { logger } from '@/lib/logger'

export type SystemParameterKey =
  | 'auto_confirm_days'
  | 'earnings_hold_hours'
  | 'refund_window_days'
  | 'dividend_period_minutes'

export interface SystemParameterDef {
  key: SystemParameterKey
  defaultValue: number
  min: number
  max: number
  description: string
  unit: string  // 'days' | 'hours' | 'minutes'
}

export const SYSTEM_PARAMETERS: Record<SystemParameterKey, SystemParameterDef> = {
  auto_confirm_days: {
    key: 'auto_confirm_days',
    defaultValue: 7,
    min: 0.0007,
    max: 30,
    description: '自动确认收货时间（发货后多久自动完成）',
    unit: 'days',
  },
  earnings_hold_hours: {
    key: 'earnings_hold_hours',
    defaultValue: 24,
    min: 0,
    max: 168,
    description: '收益到账缓冲期（确认收货后多久可提现）',
    unit: 'hours',
  },
  refund_window_days: {
    key: 'refund_window_days',
    defaultValue: 7,
    min: 0,
    max: 30,
    description: '可申请退款时间窗口（发货后多久内可退款）',
    unit: 'days',
  },
  dividend_period_minutes: {
    key: 'dividend_period_minutes',
    defaultValue: 1440,
    min: 1,
    max: 10080,
    description: '分红结算周期（订单入池后多久结算）',
    unit: 'minutes',
  },
}

// 读：拿单个配置项（带缓存，参考 business.ts 的 cache 实现）
let cache: Record<string, { value: number; time: number }> = {}
const CACHE_TTL = 60_000

export async function getSystemParameter(key: SystemParameterKey): Promise<number> {
  const cached = cache[key]
  if (cached && Date.now() - cached.time < CACHE_TTL) return cached.value

  const row = await prisma.systemConfig.findUnique({ where: { key } })
  const def = SYSTEM_PARAMETERS[key]
  let value: number
  if (!row) {
    value = def.defaultValue
  } else {
    const parsed = Number(row.value)
    value = isNaN(parsed) ? def.defaultValue : parsed
  }
  cache[key] = { value, time: Date.now() }
  return value
}

// 写：更新配置项 + 清缓存
export async function setSystemParameter(
  key: SystemParameterKey,
  value: number,
  adminId: string
): Promise<void> {
  const def = SYSTEM_PARAMETERS[key]
  if (value < def.min || value > def.max) {
    throw new Error(`${key} 必须在 ${def.min} - ${def.max} 之间`)
  }

  await prisma.$transaction(async (tx) => {
    await tx.systemConfig.upsert({
      where: { key },
      create: {
        key,
        value: String(value),
        description: def.description,
      },
      update: {
        value: String(value),
        description: def.description,
      },
    })

    await tx.operationLog.create({
      data: {
        userId: adminId,
        action: 'UPDATE',
        module: 'system_config',
        targetId: key,
        oldValue: { value: cache[key]?.value },
        newValue: { value },
        ip: null,
        userAgent: null,
      },
    })
  })

  // 清缓存
  delete cache[key]
  logger.info(`系统配置更新: ${key} = ${value}`)
}

// 一次性拿全部（给后台页面用）
export async function getAllSystemParameters(): Promise<
  Array<{ key: SystemParameterKey; value: number; def: SystemParameterDef }>
> {
  const keys = Object.keys(SYSTEM_PARAMETERS) as SystemParameterKey[]
  const values = await Promise.all(keys.map(k => getSystemParameter(k)))
  return keys.map((k, i) => ({
    key: k,
    value: values[i],
    def: SYSTEM_PARAMETERS[k],
  }))
}
```

## 3. 接口 `src/app/api/admin/system-config/parameters/route.ts`（新文件）

```typescript
// GET: 获取所有配置项
// PUT: 更新配置项（body: { key, value }）

import { NextRequest } from 'next/server'
import { prisma } from '@/lib/prisma'
import { verifyPermission } from '@/lib/utils/admin-auth'
import { errorResponse, successResponse } from '@/lib/api-response'
import {
  getAllSystemParameters,
  setSystemParameter,
  SYSTEM_PARAMETERS,
  SystemParameterKey,
} from '@/lib/config/system-parameters'

export async function GET(request: NextRequest) {
  const { user, error } = await verifyPermission(request, ['super_admin'])
  if (error || !user) return error || errorResponse('权限不足', 403)

  const params = await getAllSystemParameters()
  return successResponse({ parameters: params })
}

export async function PUT(request: NextRequest) {
  const { user, error } = await verifyPermission(request, ['super_admin'])
  if (error || !user) return error || errorResponse('权限不足', 403)

  const body = await request.json()
  const { key, value } = body as { key: SystemParameterKey; value: number }

  if (!key || !SYSTEM_PARAMETERS[key]) {
    return errorResponse('无效的配置项 key', 400)
  }
  if (typeof value !== 'number' || isNaN(value)) {
    return errorResponse('value 必须是数字', 400)
  }

  try {
    await setSystemParameter(key, value, user.id)
    return successResponse({ key, value }, '更新成功')
  } catch (err: any) {
    return errorResponse(err.message || '更新失败', 400)
  }
}
```

**注意**：本批只给 `super_admin` 权限（避免胡子上线时其他角色误改）。后续 Batch 3 角色权限重构会细化。

```typescript
'use client'

import { useState, useEffect } from 'react'

interface Parameter {
  key: string
  value: number
  def: {
    defaultValue: number
    min: number
    max: number
    description: string
    unit: string
  }
}

export default function SystemParametersPage() {
  const [params, setParams] = useState<Parameter[]>([])
  const [editing, setEditing] = useState<Record<string, string>>({})
  const [saving, setSaving] = useState<string | null>(null)

  useEffect(() => {
    fetch('/api/admin/system-config/parameters')
      .then(r => r.json())
      .then(data => setParams(data.data.parameters))
  }, [])

  const handleSave = async (key: string) => {
    const value = Number(editing[key])
    setSaving(key)
    try {
      const res = await fetch('/api/admin/system-config/parameters', {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ key, value }),
      })
      const data = await res.json()
      if (data.success) {
        // 刷新列表
        const refreshed = await fetch('/api/admin/system-config/parameters').then(r => r.json())
        setParams(refreshed.data.parameters)
        alert('更新成功')
      } else {
        alert(data.message || '更新失败')
      }
    } finally {
      setSaving(null)
    }
  }

  return (
    <div className="p-6">
      <h1 className="text-2xl font-bold mb-6">系统参数配置</h1>
      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
        {params.map(p => (
          <div key={p.key} className="bg-white rounded-lg shadow p-4">
            <div className="text-sm text-gray-500">{p.key}</div>
            <div className="text-lg font-medium mt-1">{p.def.description}</div>
            <div className="mt-3 text-3xl font-bold text-orange-600">
              {p.value} <span className="text-sm text-gray-500">{p.def.unit}</span>
            </div>
            <div className="mt-3 flex items-center gap-2">
              <input
                type="number"
                step={p.def.min < 1 ? '0.001' : '1'}
                min={p.def.min}
                max={p.def.max}
                value={editing[p.key] ?? p.value}
                onChange={e => setEditing({ ...editing, [p.key]: e.target.value })}
                className="border rounded px-2 py-1 w-32"
              />
              <span className="text-sm text-gray-500">
                ({p.def.min} - {p.def.max} {p.def.unit})
              </span>
              <button
                onClick={() => handleSave(p.key)}
                disabled={saving === p.key}
                className="ml-auto bg-orange-600 text-white px-3 py-1 rounded hover:bg-orange-700 disabled:opacity-50"
              >
                {saving === p.key ? '保存中...' : '保存'}
              </button>
            </div>
          </div>
        ))}
      </div>
    </div>
  )
}
```

import { PrismaClient } from '@prisma/client'
import { SYSTEM_PARAMETERS } from '../src/lib/config/system-parameters'

const prisma = new PrismaClient()

async function main() {
  for (const def of Object.values(SYSTEM_PARAMETERS)) {
    await prisma.systemConfig.upsert({
      where: { key: def.key },
      create: {
        key: def.key,

}
